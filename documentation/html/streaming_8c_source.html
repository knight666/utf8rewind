49 uint8_t stream_read(StreamState* state, uint8_t property)
50 {
51  /* Ensure input is available */
52 
53  if (state->src_size == 0)
54  {
55  return 0;
56  }
57 
58  /* Reset sequence after the first pass */
59 
60  if (state->filled > 0)
61  {
62  /* Check for end of data */
63 
64  if (state->filled == state->current &&
65  state->src_size <= state->last_length)
66  {
67  state->src_size = 0;
68 
69  state->index = 0;
70  state->current = 0;
71  state->filled = 0;
72 
73  return 0;
74  }
75 
76  /* Copy last peeked codepoint to new sequence */
77 
78  state->codepoint[0] = state->codepoint[state->filled - 1];
79  state->canonical_combining_class[0] = state->canonical_combining_class[state->filled - 1];
80  state->quick_check[0] = state->quick_check[state->filled - 1];
81 
82  /* New sequence always starts as stable */
83 
84  state->stable = 1;
85 
86  /* Reset buffer members */
87 
88  state->index = 0;
89  state->current = 1;
90  state->filled = 1;
91  }
92 
93  /* Read codepoints */
94 
95  while (state->filled < STREAM_SAFE_MAX)
96  {
97  /* Move the input cursor after peeking */
98 
99  if (state->last_length > 0)
100  {
101  if (state->src_size <= state->last_length)
102  {
103  break;
104  }
105 
106  state->src += state->last_length;
107  state->src_size -= state->last_length;
108  }
109 
110  /* Peek the next codepoint */
111 
112  state->last_length = codepoint_read(state->src, state->src_size, &state->codepoint[state->filled]);
113 
114  state->quick_check[state->filled] = (property != 0) ? database_queryproperty(state->codepoint[state->filled], property) : QuickCheckResult_Yes;
115  state->canonical_combining_class[state->filled] = database_queryproperty(state->codepoint[state->filled], UnicodeProperty_CanonicalCombiningClass);
116 
117  state->filled++;
118 
119  if (state->current > 0)
120  {
121  /* Sequences end on the next starter and can consist of only non-starters */
122 
123  if (state->canonical_combining_class[state->current] == 0)
124  {
125  break;
126  }
127 
128  /* Check if sequence is unstable by comparing canonical combining classes */
129 
130  if (state->stable &&
131  state->canonical_combining_class[state->current] < state->canonical_combining_class[state->current - 1])
132  {
133  state->stable = 0;
134  }
135  }
136 
137  state->current++;
138  }
139 
140  if (state->filled == STREAM_SAFE_MAX)
141  {
142  /* Insert COMBINING GRAPHEME JOINER into output */
143 
144  state->codepoint[state->filled] = 0x034F;
145  state->quick_check[state->filled] = 0;
146  state->canonical_combining_class[state->filled] = 0;
147 
148  state->filled++;
149  }
150

152 }

183 uint8_t stream_reorder(StreamState* state)
184 {
185  uint8_t i;
186  uint8_t dirty = 1;
187 
188  if (state->current == 0)
189  {
190  /* Nothing to do */
191 
192  return 0;
193  }
194 
195  /* Reorder codepoints until the entire sequence is table */
196 
197  do
198  {
199  dirty = 0;
200 
201  for (i = 1; i < state->current; i++)
202  {
203  /* Sort codepoints by canonical combining class, smallest to largest */
204 
205  if (state->canonical_combining_class[i] < state->canonical_combining_class[i - 1])
206  {
207  unicode_t swap_cp;
208  uint8_t swap_qc;
209  uint8_t swap_ccc;
210 
211  swap_cp = state->codepoint[i];
212  state->codepoint[i] = state->codepoint[i - 1];
213  state->codepoint[i - 1] = swap_cp;
214 
215  swap_qc = state->quick_check[i];

218
