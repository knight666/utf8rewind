80 uint8_t codepoint_encoded_length(unicode_t codepoint)
81 {
82  if (codepoint < 0x80)
83  {
84  /* Single byte */
85 
86  return 1;
87  }
88  else if (codepoint < 0x800)

115 {
116  uint8_t encoded_length = codepoint_encoded_length(encoded);
117  if (encoded_length == 0)

119  encoded = REPLACEMENT_CHARACTER;
120  encoded_length = 3;
121  }
122 
123  if (*target != 0)
124  {

171 uint8_t codepoint_read(const char* input, size_t inputSize, unicode_t* decoded)
172 {
173  const uint8_t* src = (const uint8_t*)input;
174 
175  if (input == 0 ||
176  inputSize == 0)
177  {
178  /* Invalid data */
179 
180  return 0;
181  }
182 
183  if (*src < 0x80)
184  {
185  /* ASCII */
186 
187  *decoded = (unicode_t)*src;
188 
189  return 1;
190  }
191  else
192  {
193  /* Multi-byte sequence */
194 
195  static const uint8_t SequenceMask[7] = { 0x00, 0x7F, 0x1F, 0x0F, 0x07, 0x03, 0x01 };
196  static const unicode_t SequenceMinimum[7] = { 0x0000, 0x0000, 0x0080, 0x0800, 0x10000, 0x0000, 0x0000 };
197 
198  size_t src_size = inputSize;
199  uint8_t src_index;
200 
201  /* Length of sequence is determined by first byte */
202 
203  uint8_t decoded_length = codepoint_decoded_length[*src];
204  if (decoded_length == 0)
205  {
206  /* Not a multi-byte sequence starter */
207 
208  *decoded = REPLACEMENT_CHARACTER;
209  decoded_length = 1;
210  }
211  else if (decoded_length >= 5)
212  {
213  /* Always an overlong sequence */
214 
215  *decoded = REPLACEMENT_CHARACTER;
216 
217  /* All bytes in the sequence must be processed */
218 
219  for (src_index = 1; src_index < decoded_length; ++src_index)
220  {
221  src++;
222 
223  /* Check if next byte is valid */
224 
225  if (src_size == 0 || /* Not enough data */
226  (*src < 0x80 || *src > 0xBF)) /* Not a continuation byte */
227  {
228  return src_index;
229  }
230 
231  src_size--;
232  }
233  }
234  else
235  {
236  /* Use mask to strip value from first byte */
237 
238  *decoded = (unicode_t)(*src & SequenceMask[decoded_length]);
239 
240  /* All bytes in the sequence must be processed */
241 
242  for (src_index = 1; src_index < decoded_length; ++src_index)
243  {
244  src++;
245 
246  /* Check if next byte is valid */
247 
248  if (src_size == 0 || /* Not enough data */
249  (*src < 0x80 || *src > 0xBF)) /* Not a continuation byte */
250  {
251  *decoded = REPLACEMENT_CHARACTER;
252 
253  return src_index;
254  }
255 
256  src_size--;
257 
258  /* Add value of continuation byte to codepoint */
259 
260  *decoded = (*decoded << 6) | (*src & 0x3F);
261  }
262 
263  /* Check for overlong sequences and surrogate pairs */
264 
265  if ((*decoded < SequenceMinimum[decoded_length] || *decoded > MAX_LEGAL_UNICODE) ||
266  (*decoded >= SURROGATE_HIGH_START && *decoded <= SURROGATE_HIGH_END) ||
267  (*decoded >= SURROGATE_LOW_START && *decoded <= SURROGATE_LOW_END))
268  {
269  *decoded = REPLACEMENT_CHARACTER;
270  }
271  }
272 
273  return decoded_length;
274  }
