59 uint8_t compose_readcodepoint(ComposeState* state, uint8_t index)
60 {
61  if (state->input->index == state->input->current &&
62  !stream_read(state->input, state->property))
63  {

81 unicode_t compose_execute(ComposeState* state)
82 {
83  uint8_t output_index;
84  uint8_t cursor_current;
85  uint8_t cursor_next;
86 
87  /* Check if input is available */
88 
89  if (state->input == 0)
90  {
91  return 0;
92  }
93 
94  /* Reset output */
95 
96  state->output->current = 0;
97 
98  /* Read first codepoint */
99 
100  if (!compose_readcodepoint(state, 0))
101  {
102  return 0;
103  }
104 
105  for (output_index = 0; output_index < state->output->current; ++output_index)
106  {
107  /* Ensure current codepoint is a starter */
108 
109  cursor_current = output_index;
110 
111  while (state->output->canonical_combining_class[cursor_current] != 0)
112  {
113  cursor_current++;
114 
115  if (cursor_current == state->output->current &&
116  !compose_readcodepoint(state, cursor_current))
117  {
118  /* Only non-starters left */
119 
120  return 1;
121  }
122  }
123 
124  /* Get next codepoint */
125 
126  cursor_next = cursor_current + 1;
127 
128  while (
129  cursor_next < state->output->current ||
130  compose_readcodepoint(state, cursor_next))
131  {
132  /*
133  Two codepoints can be composed if the current codepoint is a starter
134  and the next codepoint isn't blocked by a previous codepoint.
135  */
136 
137  if (state->output->canonical_combining_class[cursor_next] > state->output->canonical_combining_class[cursor_next - 1] || /* Can be composed based on CCC */
138  /* Quick check value can override composition block by previous codepoint */
139  (state->output->quick_check[cursor_next] != QuickCheckResult_Yes && state->output->canonical_combining_class[cursor_next - 1] == 0))
140  {
141  unicode_t composed = 0;
142 
143  /*
144  Hangul composition
145 
146  Algorithm adapted from Unicode Technical Report #15:
147  http://www.unicode.org/reports/tr15/tr15-18.html#Hangul
148  */
149 
150  if (state->output->codepoint[cursor_current] >= HANGUL_L_FIRST &&
151  state->output->codepoint[cursor_current] <= HANGUL_L_LAST)
152  {
153  /* Check for Hangul LV pair */
154 
155  if (state->output->codepoint[cursor_next] >= HANGUL_V_FIRST &&
156  state->output->codepoint[cursor_next] <= HANGUL_V_LAST)
157  {
158  unicode_t l_index = state->output->codepoint[cursor_current] - HANGUL_L_FIRST;
159  unicode_t v_index = state->output->codepoint[cursor_next] - HANGUL_V_FIRST;
160 
161  composed = HANGUL_S_FIRST + (((l_index * HANGUL_V_COUNT) + v_index) * HANGUL_T_COUNT);
162  }
163  }
164  else if (
165  state->output->codepoint[cursor_current] >= HANGUL_S_FIRST &&
166  state->output->codepoint[cursor_current] <= HANGUL_S_LAST)
167  {
168  /* Check for Hangul LV and T pair */
169 
170  if (state->output->codepoint[cursor_next] >= HANGUL_T_FIRST &&
171  state->output->codepoint[cursor_next] <= HANGUL_T_LAST)
172  {
173  unicode_t t_index = state->output->codepoint[cursor_next] - HANGUL_T_FIRST;
174 
175  composed = state->output->codepoint[cursor_current] + t_index;
176  }
177  }
178  else
179  {
180  /* Attempt to compose codepoints using the database */
181 
182  composed = database_querycomposition(
183  state->output->codepoint[cursor_current],
184  state->output->codepoint[cursor_next]);
185  }
186 
187  /* Check if composition succeeded */
188 
189  if (composed != 0)
190  {
191  /*
192  When we successfully compose two codepoints, the second must be removed
193  from the sequence. The way this is accomplished is by marking the cell
194  empty with a NUL codepoint.
195 
196  Decomposed:
197 
198  codepoint U+0044 U+0307 U+0031
199  index 0 1 2
200 
201  Composed:
202 
203  codepoint U+1E0A U+0000 U+0031
204  index 0 1 2
205 
206  If the second codepoint was at the end of the sequence, the output
207  sequence is shortened by one.
208  */
209 
210  /* Add composition to output */
211 
212  state->output->codepoint[cursor_current] = composed;
213  state->output->quick_check[cursor_current] = database_queryproperty(composed, state->property);
214  state->output->canonical_combining_class[cursor_current] = database_queryproperty(composed, UnicodeProperty_CanonicalCombiningClass);
215 
216  /* Clear next codepoint from output */
217 
218  state->output->codepoint[cursor_next] = 0;
219  state->output->quick_check[cursor_next] = 0;
220  state->output->canonical_combining_class[cursor_next] = 0;
221 
222  if (cursor_next == state->output->current - 1)
223  {
224  /* Next codepoint was at end of output */
225 
226  state->output->current--;
227  }
228 
229  /* Reset cursor to current output index */
230 
231  cursor_current = output_index;
232  cursor_next = output_index;
233  }
234  }
235  else if (
236  state->output->canonical_combining_class[cursor_next] == 0)
237  {
238  /* Attempt to compose starters, but do not read from the next sequence */
239 
240  break;
241  }
242 
243  /* Evaluate next codepoint */
244 
245  cursor_next++;
246  }
247 
248  /* Fill up "holes" left by composing codepoints not at the end of the sequence */
249 
250  if (state->output->current > 1)
251  {
252  uint8_t write_index = 0;
253  uint8_t read_index = 1;
254 
255  /*
256  We want to move valid codepoints to the left as much as possible in order to fill up
257  holes left by the composition process.
258 
259  Note that the process does not clear unused codepoints at the end, this is a small
260  optimization in order to avoid unnecessary clears. The length member is adjusted to
261  the new size.
262 
263  Before reordering:
264 
265  codepoint A B 0 0 0 D
266  index 0 1 2 3 4 5
267  length 6
268 
269  After reordering:
270 
271  codepoint A B D 0 0 D
272  index 0 1 2 3 4 5
273  length 3
274  */
275 
276  /* Evaluate all codepoints in output sequence */
277 
278  while (write_index < state->output->current)
279  {
280  /* Check if read cursor is on an empty cell */
281 
282  if (read_index < state->output->current &&
283  state->output->codepoint[read_index] == 0)
284  {
285  /* Skip all empty cells */
286 
287  while (
288  read_index < state->output->current &&
289  state->output->codepoint[read_index] == 0)
290  {
291  read_index++;
292  }
293 
294  if (read_index == state->output->current)
295  {
296  /* Reached end of data */
297 
298  break;
299  }
300 
301  /* Copy cell at read cursor to write cursor */
302 
303  state->output->codepoint[write_index] = state->output->codepoint[read_index];
304  state->output->quick_check[write_index] = state->output->quick_check[read_index];
305  state->output->canonical_combining_class[write_index] = state->output->canonical_combining_class[read_index];
306  }
307 
308  /* Move cursors */
309 
310  write_index++;
311  read_index++;
312  }
313 
314  /* Adjust length of output sequence */
315 
316  state->output->current = write_index;
317  }
318  else
319  {
320  /* Evaluated all sequences in output */
321 
322  state->input = 0;
323 
324  break;
325  }
326  }
327 
328  return 1;
329 }
