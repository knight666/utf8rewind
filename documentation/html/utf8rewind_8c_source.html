37 size_t utf8len(const char* text)
38 {
39  const char* src;
40  size_t src_index;
41  size_t length = 0;
42  size_t codepoint_length = 0;
43  size_t text_length = 0;
44  uint8_t codepoint = 0;
45 
46  if (text == 0)
47  {
48  return length;
49  }
50 
51  text_length = strlen(text);
52  if (text_length == 0)
53  {
54  return length;
55  }
56 
57  while (*text != 0)
58  {
59  codepoint = (uint8_t)*text;
60 
61  codepoint_length = codepoint_decoded_length[codepoint];
62  if (codepoint_length != 0)
63  {
64  /* Check if encoding is valid */
65 
66  src = text + 1;
67 
68  for (src_index = 1; src_index < codepoint_length; ++src_index)
69  {
70  if ((*src & 0x80) == 0)
71  {
72  /* Not a continuation byte for a multi-byte sequence */
73 
74  break;
75  }
76 
77  src++;
78  }
79  }
80  else
81  {
82  src_index = 1;
83  }
84 
85  length++;
86 
87  if (src_index > text_length)
88  {
89  break;
90  }
91 
92  text += src_index;
93  text_length -= src_index;
94  }
95 
96  return length;
97 }

99 size_t utf16toutf8(const utf16_t* input, size_t inputSize, char* target, size_t targetSize, int32_t* errors)
100 {
101  const utf16_t* src = input;
102  size_t src_size = inputSize;
103  char* dst = target;
104  size_t dst_size = targetSize;
105  size_t bytes_written = 0;
106 
107  if (input == 0 ||
108  inputSize < sizeof(utf16_t))
109  {
110  goto invaliddata;
111  }
112 
113  while (src_size > 0)
114  {
115  unicode_t codepoint = (unicode_t)*src;
116  uint8_t encoded_size;
117 
118  if (codepoint == 0)
119  {
120  break;
121  }
122 
123  if (codepoint >= SURROGATE_HIGH_START &&
124  codepoint <= SURROGATE_LOW_END)
125  {
126  utf16_t surrogate_low;
127 
128  if (codepoint > SURROGATE_HIGH_END)
129  {
130  goto unmatchedhigh;
131  }
132 
133  if (src_size < sizeof(utf16_t))
134  {
135  goto invaliddata;
136  }
137 
138  src++;
139  src_size -= sizeof(utf16_t);
140 
141  surrogate_low = *src;
142 
143  if (surrogate_low < SURROGATE_LOW_START ||
144  surrogate_low > SURROGATE_LOW_END)
145  {
146  goto unmatchedlow;
147  }
148 
149  codepoint =
150  0x10000 +
151  (surrogate_low - SURROGATE_LOW_START) +
152  ((codepoint - SURROGATE_HIGH_START) << 10);
153  }
154 
155  encoded_size = codepoint_write(codepoint, &dst, &dst_size);
156  if (encoded_size == 0)
157  {
158  goto outofspace;
159  }
160 
161  bytes_written += encoded_size;
162 
163  src++;
164  src_size -= sizeof(utf16_t);
165  }
166 
167  return bytes_written;
168 
169 invaliddata:
170  if (errors != 0)
171  {
172  *errors = UTF8_ERR_INVALID_DATA;
173  }
174  return bytes_written;
175 
176 outofspace:
177  if (errors != 0)
178  {
179  *errors = UTF8_ERR_NOT_ENOUGH_SPACE;
180  }
181  return bytes_written;
182 
183 unmatchedhigh:
184  if (errors != 0)
185  {
186  *errors = UTF8_ERR_UNMATCHED_HIGH_SURROGATE_PAIR;
187  }
188  return bytes_written;
189 
190 unmatchedlow:
191  if (errors != 0)
192  {
193  *errors = UTF8_ERR_UNMATCHED_LOW_SURROGATE_PAIR;
194  }
195  return bytes_written;
196 }

308 size_t utf8toutf16(const char* input, size_t inputSize, utf16_t* target, size_t targetSize, int32_t* errors)
309 {
310  const char* src = input;
311  size_t src_length = inputSize;
312  utf16_t* dst = target;
313  size_t dst_size = targetSize;
314  size_t bytes_written = 0;
315 
316  if (input == 0 ||
317  inputSize == 0)
318  {
319  goto invaliddata;
320  }
321 
322  if (target != 0 &&
323  targetSize < sizeof(utf16_t))
324  {
325  goto outofspace;
326  }
327 
328  while (src_length > 0)
329  {
330  unicode_t decoded;
331  uint8_t decoded_size = codepoint_read(src, src_length, &decoded);
332 
333  if (decoded <= MAX_BASIC_MULTILINGUAR_PLANE)
334  {
335  if (dst != 0)
336  {
337  if (dst_size < sizeof(utf16_t))
338  {
339  goto outofspace;
340  }
341 
342  if (decoded >= SURROGATE_HIGH_START &&
343  decoded <= SURROGATE_LOW_END)
344  {
345  *dst++ = REPLACEMENT_CHARACTER;
346  }
347  else
348  {
349  *dst++ = (utf16_t)decoded;
350  }
351 
352  dst_size -= sizeof(utf16_t);
353  }
354 
355  bytes_written += sizeof(utf16_t);
356  }
357  else
358  {
359  if (dst != 0)
360  {
361  /* Codepoint must be converted to a surrogate pair. */
362 
363  if (dst_size < sizeof(unicode_t))
364  {
365  goto outofspace;
366  }
367 
368  decoded -= 0x10000;
369  *dst++ = (decoded >> 10) + SURROGATE_HIGH_START;
370  *dst++ = (decoded & 0x3FF) + SURROGATE_LOW_START;
371 
372  dst_size -= sizeof(unicode_t);
373  }
374 
375  bytes_written += sizeof(unicode_t);
376  }
377 
378  src += decoded_size;
379  src_length -= decoded_size;
380  }
381 
382  return bytes_written;
383 
384 invaliddata:
385  if (errors != 0)
386  {
387  *errors = UTF8_ERR_INVALID_DATA;
388  }
389  return bytes_written;
390 
391 outofspace:
392  if (errors != 0)
393  {
394  *errors = UTF8_ERR_NOT_ENOUGH_SPACE;
395  }
396  return bytes_written;
397 }

469 const char* utf8seek(const char* text, const char* textStart, off_t offset, int direction)
470 {
471  size_t textLength = strlen(textStart);
472  const char* textEnd = textStart + textLength;
473 
474  switch (direction)
475  {
476 
477  case SEEK_CUR:
478  {
479  if (offset == 0)
480  {
481  return text;
482  }
483  else if (offset > 0)
484  {
485  return seeking_forward(text, textEnd, textLength, offset);
486  }
487  else
488  {
489  return seeking_rewind(textStart, text, textLength, offset);
490  }
491 
492  } break;
493 
494  case SEEK_SET:
495  {
496  if (text < textStart)
497  {
498  return text;
499  }
500 
501  return seeking_forward(textStart, textEnd, textLength, offset);
502 
503  } break;
504 
505  case SEEK_END:
506  return seeking_rewind(textStart, textEnd, textLength, -offset);
507 
508  default:
509  return text;
510 
511  }
512 }

687 size_t utf8normalize(const char* input, size_t inputSize, char* target, size_t targetSize, size_t flags, int32_t* errors)
688 {
689  char* dst = target;
690  size_t dst_size = targetSize;
691  StreamState stream[4];
692  DecomposeState decompose_state;
693  ComposeState compose_state;
694  uint8_t compatibility = (flags & UTF8_NORMALIZE_COMPATIBILITY) != 0;
695  StreamState* stream_output;
696  uint8_t finished = 0;
697  size_t bytes_written = 0;
698 
699  /*
700  Decomposition uses the following process:
701 
702  input --> stream[0] -->
703  (decompose) --> stream[1] -->
704  (accumulate) --> stream[2] -->
705  output
706 
707  The accumulation step is necessary in order to prevent buffer overflow
708  attacks.
709 
710  Composition adds another stream buffer:
711 
712  input --> stream[0] -->
713  (decompose) --> stream[1] -->
714  (accumulate) --> stream[2] -->
715  (compose) --> stream[3] -->
716  output
717 
718  Although four streaming buffers may seem excessive, they are necessary
719  for preventing allocations on the heap.
720  */
721 
722  /* Check for valid flags */
723 
724  if ((flags & (UTF8_NORMALIZE_DECOMPOSE | UTF8_NORMALIZE_COMPOSE)) == 0)
725  {
726  goto invalidflag;
727  }
728 
729  /* Validate input */
730 
731  UTF8_VALIDATE_INPUT;
732 
733  /* Initialize decomposition */
734 
735  memset(stream, 0, 4 * sizeof(StreamState));
736 
737  if (!stream_initialize(&stream[0], input, inputSize) ||
738  !decompose_initialize(&decompose_state, &stream[0], &stream[1], compatibility))
739  {
740  goto invaliddata;
741  }
742 
743  stream_output = &stream[2];
744 
745  if ((flags & UTF8_NORMALIZE_COMPOSE) != 0)
746  {
747  /* Initialize composition */
748 
749  if (!compose_initialize(&compose_state, &stream[2], &stream[3], compatibility))
750  {
751  goto invaliddata;
752  }
753 
754  stream_output = &stream[3];
755  }
756 
757  do
758  {
759  uint8_t write = 0;
760 
761  /* Accumulate decomposed input in next stream */
762 
763  if (stream[1].current > 0)
764  {
765  unicode_t* src_codepoint = stream[1].codepoint;
766  unicode_t* dst_codepoint = stream[2].codepoint + stream[2].filled;
767  uint8_t* src_qc = stream[1].quick_check;
768  uint8_t* dst_qc = stream[2].quick_check + stream[2].filled;
769  uint8_t* src_ccc = stream[1].canonical_combining_class;
770  uint8_t* dst_ccc = stream[2].canonical_combining_class + stream[2].filled;
771 
772  if ((flags & UTF8_NORMALIZE_COMPOSE) != 0)
773  {
774  uint8_t i;
775 
776  /* Update stream properties to use composition values */
777 
778  for (i = 0; i < stream[1].current; ++i)
779  {
780  *dst_qc++ = database_queryproperty(*src_codepoint, compose_state.property);
781  *dst_ccc++ = *src_ccc++;
782  *dst_codepoint++ = *src_codepoint++;
783  }
784  }
785  else
786  {
787  /* Copy directly */
788 
789  memcpy(dst_codepoint, src_codepoint, stream[1].current * sizeof(unicode_t));
790  memcpy(dst_qc, src_qc, stream[1].current * sizeof(uint8_t));
791  memcpy(dst_ccc, src_ccc, stream[1].current * sizeof(uint8_t));
792  }
793 
794  stream[2].current += stream[1].current;
795  stream[2].filled += stream[1].current;
796  }
797 
798  /* Decompose input sequence into next stream */
799 
800  finished = !decompose_execute(&decompose_state);
801  if (!finished)
802  {
803  /* Output current stream it it could overflow accumulation buffer */
804 
805  write = (stream[1].current + stream[2].filled) >= STREAM_SAFE_MAX;
806  }
807 
808  /* Reorder potentially unordered decomposed stream */
809 
810  if (!stream[1].stable)
811  {
812  stream_reorder(&stream[1]);
813  }
814 
815  /* Write stream to output when overflowing or when accumulation buffer is empty*/
816 
817  if (write ||
818  finished)
819  {
820  uint8_t i;
821 
822  /* Compose accumulation buffer */
823 
824  if ((flags & UTF8_NORMALIZE_COMPOSE) != 0 &&
825  !compose_execute(&compose_state))
826  {
827  break;
828  }
829 
830  /* Write to output buffer */
831 
832  for (i = 0; i < stream_output->current; ++i)
833  {
834  uint8_t encoded_size = codepoint_write(stream_output->codepoint[i], &dst, &dst_size);
835  if (encoded_size == 0)
836  {
837  goto outofspace;
838  }
839 
840  bytes_written += encoded_size;
841  }
842 
843  /* Reset accumulation buffer */
844 
845  stream[2].current = 0;
846  stream[2].filled = 0;
847  }
848  }
849  while (!finished);
850 
851  return bytes_written;
